## Текстовые файлы

### Чтение и запись обычных текстовых файлов
```python
# Запись текстового файла
with open('example.txt', 'w', encoding='utf-8') as f:
    f.write("Привет, мир!\nЭто пример текстового файла.")

# Чтение текстового файла
with open('example.txt', 'r', encoding='utf-8') as f:
    content = f.read()
    print(content)
```

### Работа с CSV (Comma-Separated Values)
```python
import csv

# Запись CSV файла
with open('data.csv', 'w', newline='', encoding='utf-8') as f:
    writer = csv.writer(f)
    writer.writerow(['Имя', 'Возраст', 'Город'])
    writer.writerow(['Алексей', 28, 'Москва'])
    writer.writerow(['Мария', 32, 'Санкт-Петербург'])

# Чтение CSV файла
with open('data.csv', 'r', encoding='utf-8') as f:
    reader = csv.reader(f)
    for row in reader:
        print(row)
```

## Структурированные данные

### Работа с JSON
```python
import json

data = {
    "имя": "Иван",
    "возраст": 30,
    "город": "Москва",
    "навыки": ["Python", "SQL", "Linux"]
}

# Запись JSON в файл
with open('data.json', 'w', encoding='utf-8') as f:
    json.dump(data, f, ensure_ascii=False, indent=4)

# Чтение JSON из файла
with open('data.json', 'r', encoding='utf-8') as f:
    loaded_data = json.load(f)
    print(loaded_data['имя'])
```

### Работа с XML
```python
import xml.etree.ElementTree as ET

# Создание XML
root = ET.Element("person")
name = ET.SubElement(root, "name")
name.text = "Сергей"
age = ET.SubElement(root, "age")
age.text = "25"

# Запись XML в файл
tree = ET.ElementTree(root)
tree.write('person.xml', encoding='utf-8', xml_declaration=True)

# Чтение XML из файла
tree = ET.parse('person.xml')
root = tree.getroot()
print(f"Имя: {root.find('name').text}, Возраст: {root.find('age').text}")
```

## Бинарные форматы

### Работа с изображениями (Pillow)
```python
from PIL import Image
```

### Работа с Excel (openpyxl)
```python
from openpyxl import Workbook, load_workbook

# Создание Excel файла
wb = Workbook()
ws = wb.active
ws.title = "Данные"
ws['A1'] = "Название"
ws['B1'] = "Количество"
ws.append(["Яблоки", 15])
ws.append(["Апельсины", 8])
wb.save('products.xlsx')

# Чтение Excel файла
wb = load_workbook('products.xlsx')
sheet = wb['Данные']
for row in sheet.iter_rows(values_only=True):
    print(row)
```

### Работа с PDF (PyPDF2)
```python
from PyPDF2 import PdfReader, PdfWriter

# Чтение PDF
reader = PdfReader('document.pdf')
print(f"Количество страниц: {len(reader.pages)}")
first_page = reader.pages[0]
print(first_page.extract_text())

# Создание PDF
writer = PdfWriter()
writer.add_page(first_page)
with open('first_page.pdf', 'wb') as f:
    writer.write(f)
```

## Архивы

### Работа с ZIP архивами
```python
import zipfile

# Создание ZIP архива
with zipfile.ZipFile('archive.zip', 'w') as zipf:
    zipf.write('document.pdf', arcname='docs/document.pdf')
    zipf.write('data.json', arcname='data/data.json')

# Чтение ZIP архива
with zipfile.ZipFile('archive.zip', 'r') as zipf:
    print(zipf.namelist())  # Список файлов в архиве
    zipf.extract('data/data.json', 'extracted_data')
```

## Базы данных

### Работа с SQLite
```python
import sqlite3

# Создание и подключение к базе данных
conn = sqlite3.connect('database.db')
cursor = conn.cursor()

# Создание таблицы
cursor.execute('''CREATE TABLE IF NOT EXISTS users
               (id INTEGER PRIMARY KEY, name TEXT, age INTEGER)''')

# Вставка данных
cursor.execute("INSERT INTO users (name, age) VALUES (?, ?)", ('Анна', 28))
conn.commit()

# Чтение данных
cursor.execute("SELECT * FROM users")
for row in cursor.fetchall():
    print(row)

conn.close()
```

## Специализированные форматы

### Работа с YAML
```python
import yaml

data = {
    'server': {
        'host': 'localhost',
        'port': 8080,
        'ssl': True
    }
}

# Запись YAML
with open('config.yml', 'w') as f:
    yaml.dump(data, f, default_flow_style=False, allow_unicode=True)

# Чтение YAML
with open('config.yml', 'r') as f:
    config = yaml.safe_load(f)
    print(config['server']['host'])
```

### Работа с HDF5 (для научных данных)
```python
import h5py

# Запись данных
with h5py.File('data.h5', 'w') as f:
    dataset = f.create_dataset('temperatures', data=[20.5, 21.3, 19.8])
    dataset.attrs['units'] = 'Celsius'

# Чтение данных
with h5py.File('data.h5', 'r') as f:
    data = f['temperatures'][:]
    print(data)
    print(f['temperatures'].attrs['units'])
```
